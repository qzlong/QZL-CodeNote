##### 3.给定一个字符串，请你找出其中不含有重复字符的 最长子串 的长度。
> 输入: s = "abcabcbb"  
> 输出: 3   
> 解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。  
***
##### 解法一：利用HashTable求解
```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char,int> hashtable;
        int len = 0;
        int maxLen = 0;
        for(int i=0;i<s.size();++i){
            auto it = hashtable.find(s[i]);
            if(it==hashtable.end()){
                hashtable[s[i]]=i;
                len++;
                if(len>maxLen) 
                    maxLen=len;
            }else{
                int t=i-it->second;
                hashtable[s[i]]=i;
                if(t<=len){
                    len = t;
                }else{
                    len++;
                    if(len>maxLen)
                        maxLen=len;
                }
            }
        }
        return maxLen;
    }
};
```
##### 解法二：滑动串口（双指针）
```C++
class Solution{
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> uset;//使用集合来判断是否有重复的数
        int rk=-1,ans=0;
        for(int i=0;i<s.size();++i){
            if(i!=0) //每次循环将左指针移动一位
                uset.erase(s[i-1]);
            while(rk+1<s.size() && !uset.count(s[rk+1])) //判断右指针指向字符的下一个字符是否在集合中存在
                uset.insert(s[++rk]);
            ans = max(ans,rk-i+1);
        }
        return ans;
    }
};
```
