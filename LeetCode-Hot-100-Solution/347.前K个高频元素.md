### 347.前K个高频元素
> 给定一个非空的整数数组，返回其中出现频率前 k 高的元素。
***
##### 解法一：哈希表+最小堆
```c++
class Solution {
private:
    using v_t = pair<int,int>;
    static bool compare(const v_t &lh,const v_t &rh){
        return lh.second > rh.second;
    }
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int,int> hashMap;
        for(auto& num:nums){
            hashMap[num]++;
        }
        priority_queue<v_t,vector<v_t>,decltype(&compare)> p(compare);
        for(auto &[num,count]:hashMap){
            if(p.size()<k){
                p.emplace(num,count);
            }else{
                if(p.top().second<count){
                    p.pop();
                    p.emplace(num,count);
                }
            }
        }
        vector<int> ret;
        while(!p.empty()){
            ret.push_back(p.top().first);
            p.pop();
        }
        return ret;
    }
};
```
