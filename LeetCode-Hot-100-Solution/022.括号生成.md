### 22.括号生成
> 数字n代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且有效的括号组合
***
##### 解法一：回溯算法
```c++
class Solution {
private:
    vector<string> ret;
    int size;
    void backtrace(string& output,int l,int r){
        if(l==size&&r==size){
            ret.emplace_back(output);
            return;
        }
        if(l<size){
            output+="(";
            backtrace(output,l+1,r);
            output.pop_back();
        }
        //保证括号有效
        if(r<l){
            output+=")";
            backtrace(output,l,r+1);
            output.pop_back();
        }
    }
public:
    vector<string> generateParenthesis(int n) {
        size = n;
        string s;
        backtrace(s,0,0);
        return ret;
    }
};
```
